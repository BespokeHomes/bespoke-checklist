function isComplete(tab,idx){

  const item=DATA[tab][idx];

  const k=tab+'|'+idx;

  const t=st.taskCache[st.proj?.id]?.[k];

  if(!t)return false;

  if(item.mode==='checkbox')return!!t.checked;

  if(item.mode==='construction'){

    if(item.o==='g')return false;

    if(item.o==='i')return false;

    return t.status==='complete';

  }

  if(item.mode==='permits')return t.status==='completed';

  return t.status==='complete'||t.status==='completed';

}


function countTab(tab){

  const items=DATA[tab];

  let done=0,total=0;

  items.forEach((item,idx)=>{

    if(item.mode==='construction'&&(item.o==='g'||item.o==='i'))return;

    total++;

    if(isComplete(tab,idx))done++;

  });

  return{done,total};

}

async function loadTasks(projId){

  if(!projId)return;

  if(st.taskCache[projId])return;

  const{data}=await sb.from('bh_tasks').select('*').eq('project_id',projId);

  const map={};

  (data||[]).forEach(t=>map[t.task_key]=t);

  st.taskCache[projId]=map;

}


async function upsertTask(projId,taskKey,updates){

  const existing=st.taskCache[projId]?.[taskKey];

  const row={

    project_id:projId,

    task_key:taskKey,

    updated_by:st.user.initials,

    updated_by_name:st.user.name,

    updated_at:new Date().toISOString(),

    ...updates

  };

  if(existing){

    await sb.from('bh_tasks').update(row).eq('project_id',projId).eq('task_key',taskKey);

  }else{

    await sb.from('bh_tasks').insert(row);

  }

  if(!st.taskCache[projId])st.taskCache[projId]={};

  st.taskCache[projId][taskKey]={...existing,...row};

}

          ${showStamp?`<span class="stav">${t.updated_by}</span><span class="stnm">${t.updated_by_name||t.updated_by}</span><span class="stdt">· ${t.date_completed||fd()}</span><span class="stpill ${sv}">${sv.replace(/-/g,' ')}</span>`:''}

        </div>

        <div class="na${hasN?' open':''}" id="na${item.idx}">

          <textarea class="nta" placeholder="Add a note..." onchange="saveNote('${tab}',${item.idx},this.value)">${t.note||''}</textarea>

        </div>

      </div><div class="tc">`;


      if(isGC){

        h+=`<span class="gc-lbl">GC</span>`;

      }else if(isInsp){

        h+=`<select class="ss ${sv||'not-started'}" onchange="doSt('${tab}',${item.idx},this.value)">

          <option value="not-started" ${!sv||sv==='not-started'?'selected':''}>—</option>

          <option value="scheduled" ${sv==='scheduled'?'selected':''}>Scheduled</option>

          <option value="passed" ${sv==='passed'?'selected':''}>Passed</option>

          <option value="corrections" ${sv==='corrections'?'selected':''}>Corrections</option>

        </select>`;

      }else if(mode==='construction'){

        // Matthew/Cody dropdown

        h+=`<select class="ss ${sv||'not-started'}" onchange="doSt('${tab}',${item.idx},this.value)">

          <option value="not-started" ${!sv||sv==='not-started'?'selected':''}>Not Started</option>

          <option value="in-progress" ${sv==='in-progress'?'selected':''}>In Progress</option>

          <option value="complete" ${sv==='complete'?'selected':''}>Complete</option>

          <option value="n-a" ${sv==='n-a'?'selected':''}>N/A</option>

        </select>`;

      }

      // Checkbox-only tabs have no dropdown


      // Date field for Acquisition

      if(item.hasDate){

        h+=`<input type="date" class="date-inp" value="${t.date_completed||''}" onchange="saveDate('${tab}',${item.idx},this.value)" title="Date completed">`;

      }


      // Note button — not on GC rows

      if(!isGC){

        h+=`<button class="nb${hasN?' on':''}" onclick="togN(${item.idx})" id="nb${item.idx}">${hasN?'📝':'+ Note'}</button>`;

      }

      h+=`</div></div>`;

    });

  });

  return h;

}

async function doSt(tab,idx,value){

  if(!st.proj)return;

  const k=tab+'|'+idx;

  const item=DATA[tab][idx];

  const isGCrow=item.o==='g'||item.o==='i';

  const updates={status:value};

  const blank=!value||value==='not-started'||value==='not-present'||value==='incomplete';

  if(!blank){

    updates.date_completed=fd();

    if(!isGCrow){updates.updated_by=st.user.initials;updates.updated_by_name=st.user.name;}

    addLog('update',`"${item.t}" → ${value} in ${tab}`);

  }else{

    updates.date_completed=null;

    if(!isGCrow){updates.updated_by='';updates.updated_by_name='';}

  }

  await upsertTask(st.proj.id,k,updates);


  // Update UI inline

  const sel=document.querySelector(`#r${idx} select`);if(sel)sel.className=`ss ${value}`;

  const row=$('r'+idx);

  if(row&&!isGCrow){

    row.className='tr';

    if(isComplete(tab,idx))row.classList.add('done');

  }

  updStamp(tab,idx);

  updProg();

}


async function doChk(tab,idx,checked){

  if(!st.proj)return;

  const k=tab+'|'+idx;

  const item=DATA[tab][idx];

  const updates={checked,status:checked?'checked':'default'};

  if(checked){updates.updated_by=st.user.initials;updates.updated_by_name=st.user.name;updates.date_completed=fd();addLog('checked',`Checked "${item.t}" in ${tab}`);}

  else{updates.updated_by='';updates.updated_by_name='';updates.date_completed=null;}

  await upsertTask(st.proj.id,k,updates);

  const row=$('r'+idx);if(row)row.classList.toggle('checked',checked);

  updProg();

}

function updStamp(tab,idx){

  const sd=$('st'+idx);if(!sd)return;

  const k=tab+'|'+idx;

  const t=st.taskCache[st.proj?.id]?.[k]||{};

  const sv=t.status||'';

  const blank=!sv||sv==='not-started'||sv==='default';

  const item=DATA[tab][idx];

  const isGC=item.mode==='construction'&&(item.o==='g'||item.o==='i');

  if(t.updated_by&&!blank&&!isGC){

    sd.style.display='flex';

    sd.innerHTML=`<span class="stav">${t.updated_by}</span><span class="stnm">${t.updated_by_name||t.updated_by}</span><span class="stdt">· ${t.date_completed||''}</span><span class="stpill ${sv}">${sv.replace(/-/g,' ')}</span>`;

  }else{sd.style.display='none';sd.innerHTML='';}

}


function updProg(){

  if(!st.proj)return;

  let total=0,done=0;

  TABS.forEach(tab=>{const c=countTab(tab);total+=c.total;done+=c.done;});

  const pct=total>0?Math.round((done/total)*100):0;

  const pf=$('pf'),pp=$('pp'),pc=$('pc');

  if(pf)pf.style.width=pct+'%';if(pp)pp.textContent=pct+'%';if(pc)pc.textContent=done+' / '+total;

  const tab=st.tab;const{done:td,total:tt}=countTab(tab);

  const dc=$('dc'),rm=$('rm');

  if(dc)dc.textContent=td;if(rm)rm.textContent=tt-td;

}
